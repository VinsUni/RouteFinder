CREATE TABLE `route_descriptions` (`ROUTE_ID_URL` VARCHAR(100) NOT NULL,   
`ROUTE_DESCRIPTION` TEXT NOT NULL,   PRIMARY KEY (`ROUTE_ID_URL`) USING BTREE,  
  UNIQUE KEY `UNI_ROUTE_ID_URL` (`ROUTE_ID_URL`) USING BTREE ) ENGINE=InnoDB DEFAULT CHARSET=utf8;

  
  
  CREATE TABLE `route_sentenses` (`SENTENSE_ID` MEDIUMINT NOT NULL AUTO_INCREMENT,   
`SENTENSES` TEXT NOT NULL, `ROUTE_ID_URL` VARCHAR(100) NOT NULL, `ROUTE_DESCRIPTION_POSITION` INT,
  PRIMARY KEY (`SENTENSE_ID`) USING BTREE ) 
  ENGINE=InnoDB DEFAULT CHARSET=utf8;
  
  
    
  CREATE TABLE `route_landmarks` (`ID` MEDIUMINT NOT NULL AUTO_INCREMENT,   
`LANDMARK` TEXT NOT NULL, `SENTENSE_IDS` TEXT NOT NULL, 
  PRIMARY KEY (`ID`) USING BTREE ) 
  ENGINE=InnoDB DEFAULT CHARSET=utf8;
  
  
      
  CREATE TABLE `route_landmarks_graph` (`ID` MEDIUMINT NOT NULL AUTO_INCREMENT,   
`LANDMARK` TEXT NOT NULL, `ADJACENCY_LANDMARKS` TEXT NOT NULL, `SENTENSE_IDS` TEXT NOT NULL,
  PRIMARY KEY (`ID`) USING BTREE ) 
  ENGINE=InnoDB DEFAULT CHARSET=utf8;
